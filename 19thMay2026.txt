use vitcampus;
select * from worker;

select first_name,department,salary from worker
where department="admin" and salary >=100000;
select first_name,department,salary from worker
where department="admin" or salary >=100000;

select first_name,department,salary from worker
where worker_id in (1,6,7);
select first_name,department,salary from worker
where worker_id not in (1,6,7);

select * from worker
where salary between 110000 and 300000;
select * from worker
where salary between 10000 and 100000
and department not in ("hr","account");

-- LIKE Operators
select * from worker 
where first_name like "__i%";
select * from worker
where first_name like "V%e_";
select * from worker
where first_name like "_i__a%";
select * from worker
where first_name like "_m_t%";
select * from worker
where first_name like "_a%__";
select * from worker
where first_name like "__h__i%";

-- Aggregate Functions
select min(salary) from worker
where department="admin";
select count(worker_id) from worker
where department="hr";
select sum(salary) from worker
where department="account";
select avg(salary) from worker
where department="hr";

select min(salary) from worker
where department="admin" 
and first_name not like "a%";

-- UNION Operator
select salary from worker
union
select salary from worker;
select salary, worker_id from worker
union all
select salary, worker_id from worker;

-- Alias = Temporary name for column
select worker_id as RollNumber from worker;
select distinct(department) from worker;		-- unique values

-- CASE Statement
select worker_id,first_name,
case 
	when salary > 300000 then "Rich"
	when salary >= 100000 and salary < 299999 then "Middle" 
	else "Poor"
end
as status
from worker;

-- VIEW Function
create view minSalaryAdmin as select min(salary) from worker
where department="admin";
select * from minsalaryadmin;
create view maxSalaryAdmin as select max(salary) from worker
where department="admin";
select * from maxsalaryadmin;
